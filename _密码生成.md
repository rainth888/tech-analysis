
# 随机密码生成
# powershell
在powershell下，用一行脚本，生成设定位数（比如32位）随机密码，必要要有大小写字母数字特殊字符，排除'O0l1+/='。可以设定一次出得数量，比如10个。
$s='ABCDEFGHIJKLMNPQRSTUVWXYZabcdefghijkmnopqrstuvwxyz23456789!@#$%^*()_-.'; 1..10 | % { -join (1..32 | % { $s[(Get-Random $s.Length)] }) }
$chars = 'abcdefghijkmnopqrstuvwxyzABCDEFGHIJKLMNPQRSTUVWXYZ23456789!@#$%^&*()_-[]{}<>?,.~`|":;' -split '' | Where-Object {$_}; 1..5 | ForEach-Object { do { $p = -join (Get-Random -InputObject $chars -Count 32) } until ($p -match '(?=.*[a-z])(?=.*[A-Z])(?=.*[0-9])(?=.*[!@#\$%\^&\*\(\)_\-\{\}\[\]<>?,\.~`\|":;])'); $p }
1..10 | % { -join ( [char[]]('ABCDEFGHIJKLMNPQRSTUVWXYZabcdefghijkmnopqrstuvwxyz23456789!@#$%^&*()_\-[]{}\|;:,.<>?') | Sort-Object {Get-Random} | Select-Object -First 32 ) }

# Ubuntu
在Ubuntu18.04下，用一行脚本，生成设定位数（比如32位）随机密码，必要要有大小写字母数字特殊字符，排除'O0l1+/='。可以设定一次出得数量，比如5个、10个。
for i in {1..5}; do < /dev/urandom tr -dc 'A-HJ-NP-Za-kmnp-z2-9!@#$%^&*()_\-\[\]{}<>?,.:~`|";' | head -c32; echo; done

count=10 len=32 && tr -dc 'A-Za-z0-9!@#$%^&*()_\-[]{}\\|;:,.<>?' < /dev/urandom | fold -w "$len" | head -n "$count"


方法六：使用 openssl 生成
for i in {1..5}; do openssl rand -base64 48 | tr -dc 'A-HJ-NP-Za-kmnp-z2-9!@#$%^&*()_\-\[\]{}<>?,.:~`|";' | head -c 32; echo; done
方法七：使用 pwgen 工具（需安装）
sudo apt-get install -y pwgen > /dev/null 2>&1; for i in {1..5}; do pwgen -cyns 32 1 | tr -d 'O0l1+/='; done
